    /**
     * Add a new group
     *
     * @method add
     * @async
     * @param {Object}     params
     * @param {string}     params.token - auth token
     * @param {string}     params.label - name of the group
     * 
     * @return {Object}    return
     * @return {uuid}      return.groupId - generated uuid of the group
     * @return {uuid}      return.userId - uuid of the user as extracted from the token
     * @return {string}    return.label - returned parameter
     * @return {string}    return.role - granted role: 'admin'
     * 
     */
    async add({ token, label}) {
        const groupId = uuid();
        const { payload } = this.db.encryption.verify({ token });
        const user = payload?.user;
        const key = this.db.encryption.getHash(user.mail);
        const role = "admin";
        // add group
        const Group = db.getGroupInterface();
        await Group.add({ groupId, data: { label }});
        // add grant
        const Grants = db.getGrantsInterface();
        await Grants.add({ groupId, entityId: user.id, token: { groupId, ownerId: userId, role }});
        // add relation
        const Relation = db.getRelationInterface();
        await Relation.add({ key, groupId, data: { label, role }});
        return {
            groupId,
            label,
            userId: user.id,
            role
        }
    }
